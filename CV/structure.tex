%----------------------------------------------------------------------------------------
%	JOB COMMAND
%----------------------------------------------------------------------------------------

\newcommand{\job}[6]{
\begin{tabbing}
\hspace{2.5cm} \= \kill
\textbf{#1} \> \href{#4}{#3} \\
\textbf{#2} \>\+ \textit{#5} \\
\begin{minipage}{\smallertextwidth}
\vspace{1mm}
#6
\end{minipage}
\end{tabbing}
\vspace{1mm}
}

%	SKILL GROUP COMMAND

\newcommand{\skillgroup}[2]{

\end{minipage}
\end{tabbing}
}
